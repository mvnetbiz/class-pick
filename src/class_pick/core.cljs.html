 49 (defn class-list-view [class owner]
 50   (reify
 51     om/IRenderState
 52     (render-state [this {:keys [delete]}]
 53             (dom/tr nil
 54                     (dom/td nil (:crn class))
 55                     (dom/td nil (:title class))
 56                     (dom/td nil (:days class))
 57                     (dom/td nil (:time_start class))
 58                     (dom/td nil (:time_end class))
 59                     (dom/button #js {:onClick (fn [e] (put! delete @class))}
 60                                 "Delete")))))

 78 (defn add-class [data owner]
 79   (let [new-crn (-> (om/get-node owner "new-crn")
 80                       .-value)
 81         new-title (-> (om/get-node owner "new-title")
 82                       .-value)
 83         new-days (-> (om/get-node owner "new-days")
 84                      .-value)
 85         new-start (-> (om/get-node owner "new-start")
 86                       .-value)
 87         new-end (-> (om/get-node owner "new-end")
 88                     .-value)]
 89     (when (not (or (empty? new-crn)
 90                    (empty? new-title)
 91                    (empty? new-days)
 92                    (empty? new-start)
 93                    (empty? new-end)))
 94       (om/transact! data :classes #(conj % {:crn new-crn
 95                                             :title new-title
 96                                             :days new-days
 97                                             :time_start new-start
 98                                             :time_end new-end
 99                                             :state :unpicked}))))
100   ;;DEBUG THING
101   ;;PRINT APP STATE ON CLICK OF ADD CLASS BUTTON
102   (println data)
103   ;;COMMENT

106 (defn classes-view [data owner]
107   (reify
108     om/IInitState
109     (init-state [_]
110                 {:delete (chan)})
111     om/IWillMount
112     (will-mount [_]
113                 (let [delete (om/get-state owner :delete)]
114                   (go (loop []
115                         (let [class (<! delete)]
116                           (om/transact! data :classes
117                                         (fn [xs] (vec (remove #(= class %) xs))))
118                           (recur))))))
119     om/IRenderState
120     (render-state [this state]
121                   (dom/div nil
122                            (dom/h2 nil "Class Editor")
123                            (dom/div nil
124                                     (dom/table nil
125                                            (dom/tr nil
126                                                    (dom/th nil "CRN")
127                                                    (dom/th nil "Title")
128                                                    (dom/th nil "Days")
129                                                    (dom/th nil "Start")
130                                                    (dom/th nil "End")
131                                                    (dom/th nil "Add Class"))
132                                             (dom/tr nil
133                                                     (dom/td nil (dom/input #js {:type "text" :ref "new-crn"}))
134                                                     (dom/td nil (dom/input #js {:type "text" :ref "new-title"}))
135                                                     (dom/td nil (dom/input #js {:type "text" :ref "new-days"}))
136                                                     (dom/td nil (dom/input #js {:type "text" :ref "new-start"}))

155 (defn hour-str-to-minutes [str]
156   (-> str
157       (#(string/split % ":"))
158       (#(let [hr (js/parseInt (nth % 0))
159              mt (js/parseInt (nth % 1))]
160          (+ (* hr 60) mt)))))

163 (defn hour-conflicts [beg1 end1 beg2 end2]
164   (let [s1 (hour-str-to-minutes beg1)
165         e1 (hour-str-to-minutes end1)
166         s2 (hour-str-to-minutes beg2)
167         e2 (hour-str-to-minutes end2)]
168     (println "s1 " s1 " e1 " e1 " s2 " s2 " e2 " e2)
169     (or (and (>= s1 s2) (<= s1 e2))
170              (and (>= s2 s1) (<= s2 e1)))))

172 (defn check-conflicts [class1 class2]
173   (let [day1 (:days class1)
174         day2 (:days class2)
175         beg1 (:time_start class1)
176         beg2 (:time_start class2)
177         end1 (:time_end class1)
178         end2 (:time_end class2)]
179     (and (day-conflicts day1 day2)
180          (hour-conflicts beg1 end1 beg2 end2))))

182 (defn toggle-check [class eq]
183   (let [theday (:days eq)]
184     (if (= class eq)
185       (assoc class :state ({:picked :unpicked :unpicked :picked} (:state class)))
186       (if (check-conflicts class eq)
187         (assoc class :state ({:conflicts :unpicked :unpicked :conflicts} (:state class)))
188         class))))

190 (defn picker-view [data owner]
191   (reify
192     om/IInitState
193     (init-state [_]
194                 {:check (chan)})
195     om/IWillMount
196     (will-mount [_]
197                 (let [check (om/get-state owner :check)]
198                   (go (loop []
199                         (let [class (<! check)]
200                           (om/transact! data :classes
201                                         (fn [m]
202                                           (vec (map #(toggle-check % class) m))))
203                           (recur))))))
204     om/IRenderState
205     (render-state [this state]
206                   (dom/div nil
207                            (dom/h2 nil "Class Picker")
208                            (apply dom/table nil
209                                   (om/build-all class-picker-view (:classes data)
210                                                 {:init-state state}))))))
